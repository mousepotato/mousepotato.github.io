package com.anotherbug;

public class PrintStar {

	public static void main(String[] args) {

		// test1
		for (int i = 1; i <= 5; i++) {
			for (int j = 1; j <= i; j++) {
				System.out.print("*");
			}
			System.out.println();
		}

		// test2
		for (int i = 1; i <= 5; i++) {
			for (int j = 0; j <= 5 - i; j++) {
				System.out.print("*");
			}
			System.out.println();
		}

		// test3
		for (int i = 1; i <= 5; i++) {
			for (int j = 1; j <= 5 - i; j++) {
				System.out.print(" ");
			}

			for (int j = 1; j <= i; j++) {
				System.out.print("*");
			}
			System.out.println();
		}

		// test4
		for (int i = 1; i <= 5; i++) {

			for (int j = 1; j <= i - 1; j++) {
				System.out.print(" ");

			}
			for (int j = 0; j <= 5 - i; j++) {
				System.out.print("*");
			}

			System.out.println();
		}

		// test5
		for (int i = 1; i <= 5; i++) {
			for (int j = 0; j < 5 - i; j++) {
				System.out.print(" ");
			}
			for (int j = 1; j <= 2 * i - 1; j++) {
				System.out.print("*");
			}

			System.out.println();
		}

		// test6
		for (int i = 1; i <= 5; i++) {
			for (int j = 1; j <= i - 1; j++) {
				System.out.print(" ");
			}
			for (int j = 1; j <= 2 * (5 - i) + 1; j++) {
				System.out.print("*");
			}
			System.out.println();
		}

	}
}
